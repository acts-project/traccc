  ${TRACCC_CUDA_SEEDING_SOURCE_DIR}/counting_grid_capacities.cu
  ${TRACCC_CUDA_SEEDING_SOURCE_DIR}/populating_grid.cu  
  ${TRACCC_CUDA_SEEDING_SOURCE_DIR}/doublet_counting.cu

  --expt-relaxed-constexpr>)

  #-fmad=false>)
